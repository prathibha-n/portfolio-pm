Full schema (mock)
-- Core tables (90-day rolling window) drivers( driver_id UUID PK, city VARCHAR, -- 'BLR','MUM','DEL','HYD','CHN' vehicle_year INT, -- manufacturing year vehicle_type VARCHAR, -- 'petrol','diesel','cng','ev' onboarding_date DATE, is_active BOOLEAN ) trips( trip_id UUID PK, driver_id UUID FK → drivers, trip_date DATE, trip_count INT, -- trips completed that day avg_dist_km FLOAT, -- avg distance per trip that day city_centre_pct FLOAT, -- share of trips in central zones (0–1) peak_hour_pct FLOAT -- share of trips in 7–10am / 5–9pm ) cancellations( cancel_id UUID PK, driver_id UUID FK → drivers, cancel_date DATE, reason VARCHAR, -- 'low_battery','no_charge_nearby', -- 'passenger_no_show','traffic','other' trip_accepted BOOLEAN -- did driver accept before cancelling? ) charging_events( event_id UUID PK, driver_id UUID FK → drivers, event_date DATE, charge_duration_min INT, charge_location VARCHAR -- 'home','public_fast','public_slow','workplace' )
Why these tables matter
→trips — daily granularity lets you compute rolling 90-day averages without blowing up the join. trip_count per day is more reliable than counting trip_id rows (drivers occasionally have data gaps).
→cancellations.reason — the critical column. Battery-related cancels ('low_battery', 'no_charge_nearby') are your leading indicator of suppressed EV intent — drivers who would convert but fear being stranded.
→charging_events — used in the risk model to check whether existing EV drivers cluster charges at specific location types. Informs hub placement (deliverable 3).
Full annotated query
WITH -- Step 1: Aggregate driver behaviour over last 90 days driver_behaviour AS ( SELECT d.driver_id, d.city, d.vehicle_type, (EXTRACT(YEAR FROM CURRENT_DATE) - d.vehicle_year) AS vehicle_age_yrs, DATEDIFF(CURRENT_DATE, d.onboarding_date) / 30.0 AS tenure_months, -- Trip volume and pattern SUM(t.trip_count) AS total_trips_90d, AVG(t.trip_count) AS avg_daily_trips, AVG(t.avg_dist_km) AS avg_trip_dist_km, AVG(t.city_centre_pct) AS city_centre_share, AVG(t.peak_hour_pct) AS peak_hour_share, COUNT(DISTINCT t.trip_date) AS active_days_90d FROM drivers d JOIN trips t USING (driver_id) WHERE t.trip_date >= CURRENT_DATE - 90 AND d.is_active = TRUE AND d.vehicle_type != 'ev' -- exclude already-converted drivers GROUP BY 1,2,3,4,5 ), -- Step 2: Build cancellation risk signal cancel_signals AS ( SELECT driver_id, -- Raw battery cancel count COUNT(CASE WHEN reason IN ('low_battery','no_charge_nearby') AND trip_accepted = TRUE THEN 1 END) AS battery_cancels, -- All accepted-trip cancels (denominator for rate) COUNT(CASE WHEN trip_accepted = TRUE THEN 1 END) AS total_accepted_cancels, -- Battery cancel rate among accepted trips COUNT(CASE WHEN reason IN ('low_battery','no_charge_nearby') AND trip_accepted = TRUE THEN 1 END) * 1.0 / NULLIF(COUNT(CASE WHEN trip_accepted = TRUE THEN 1 END), 0) AS battery_cancel_rate FROM cancellations WHERE cancel_date >= CURRENT_DATE - 90 GROUP BY 1 ), -- Step 3: Score and tier each driver scored AS ( SELECT b.*, COALESCE(c.battery_cancels, 0) AS battery_cancels, COALESCE(c.battery_cancel_rate, 0) AS battery_cancel_rate, -- Tier logic: high-value conversion targets first CASE WHEN b.vehicle_age_yrs >= 4 AND b.avg_daily_trips >= 8 AND b.city_centre_share >= 0.60 AND b.tenure_months >= 6 THEN 'tier_1_high_value' WHEN b.vehicle_age_yrs >= 2 AND b.avg_daily_trips >= 5 AND b.tenure_months >= 3 THEN 'tier_2_mid' ELSE 'tier_3_nurture' END AS ev_tier, -- Economic proxy: trips × avg dist = daily km driven -- Higher km → faster subsidy payback for Uber ROUND(b.avg_daily_trips * b.avg_trip_dist_km, 1) AS daily_km_proxy, -- Urgency flag: high battery cancels = frustrated, primed for EV pitch CASE WHEN c.battery_cancel_rate > 0.05 THEN TRUE ELSE FALSE END AS high_battery_anxiety FROM driver_behaviour b LEFT JOIN cancel_signals c USING (driver_id) ) -- Step 4: City-level rollup for the incentive budget model SELECT ev_tier, city, COUNT(*) AS driver_count, ROUND(AVG(avg_daily_trips), 1) AS avg_trips_day, ROUND(AVG(daily_km_proxy), 1) AS avg_daily_km, ROUND(AVG(vehicle_age_yrs), 1) AS avg_vehicle_age, ROUND(AVG(battery_cancel_rate) * 100, 2) AS avg_battery_cancel_pct, SUM(CASE WHEN high_battery_anxiety THEN 1 END) AS high_anxiety_drivers, ROUND(AVG(city_centre_share) * 100, 1) AS avg_city_centre_pct FROM scored GROUP BY 1, 2 ORDER BY CASE ev_tier WHEN 'tier_1_high_value' THEN 1 WHEN 'tier_2_mid' THEN 2 ELSE 3 END, driver_count DESC;
